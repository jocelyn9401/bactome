42  def geometricMean(self):
43  """
44  Calculates the geometric mean of the data
45  """
46  mult = 1.0
47  one_over_n = 1.0 / len(self.data)
48  for item in self.data: mult = mult * math.pow(item, one_over_n)
49  return mult

51  def harmonicMean(self):
52  """
53  Calculates the harmonic mean of the data
54  """
55  sum = 0.000001
56  for item in self.data:
57  if item != 0: sum = sum + 1.0/item
58  else: sum = sum + 1.0/0.001
59  return len(self.data) / sum

61  def arithmeticMean(self):
62  """
63  Returns the arithematic mean of the data
64  """
65  sum = 0
66  for item in self.data: sum = sum + item
67  return sum / float(len(self.data))

69  def moment(self, moment=1):
70  """
71  Calculates the nth moment about the mean for the data
72  """
73  if moment == 1:
74  return 0.0
75  else:
76  mn = self.arithmeticMean()
77  n = len(self.data)
78  s = 0
79  for x in self.data:
80  s = s + (x - mn) ** moment
81  return s / float(n)

164  Calculates covariance using the formula: Cov(xy) = E(xy) - E(x)E(y)
165  """
166  sname = self.listSamples()
167  if self.sample[sname[0]].data == self.sample[sname[1]].data: return 1.0

190  error_x = [self.sample[sname[0]].data[i] - mean_x
191  for i in range(slen)]
192  error_y = [self.sample[sname[1]].data[i] - mean_y
193  for i in range(slen)]
194  gradient = sum([error_x[index] * error_y[index]
195  for index in range(len(error_x))]) / \
196  sum([error_x[index] * error_x[index]
197  for index in range(len(error_x))])
198  intercept = mean_y - (gradient * mean_x)
199  return (gradient, intercept)

201  def pearson(self):
202  """
203  Calculates the Pearson's product-moment coefficient by the formula
204 
205  (N * sum_xy) - (sum_x * sum_y)
206  --------------------------------------------------------------
207  ((N * sum_x2 - (sum_x)**2) * (N * sum_y2 - (sum_y)**2)) ** 0.5
208  """
209  sname = self.listSamples()
210  if self.sample[sname[0]].rowcount == self.sample[sname[1]].rowcount:
211  slen = self.sample[sname[0]].rowcount
212  elif self.sample[sname[0]].rowcount > self.sample[sname[1]].rowcount:
213  slen = self.sample[sname[1]].rowcount
214  else: slen = self.sample[sname[0]].rowcount
215  sum_x = summation([self.sample[sname[0]].data[i]
216  for i in range(slen)])
217  sum_x2 = summation([self.sample[sname[0]].data[i] * \
218  self.sample[sname[0]].data[i]
219  for i in range(slen)])
220  sum_y = summation([self.sample[sname[1]].data[i]
221  for i in range(slen)])
222  sum_y2 = summation([self.sample[sname[1]].data[i] * \
223  self.sample[sname[1]].data[i]
224  for i in range(slen)])
225  sum_xy = summation([self.sample[sname[0]].data[i] * \
226  self.sample[sname[1]].data[i]
227  for i in range(slen)])
228  numerator = (slen * sum_xy) - (sum_x * sum_y)
229  denominator_x = (slen * sum_x2) - (sum_x * sum_x)
230  denominator_y = (slen * sum_y2) - (sum_y * sum_y)
231  return float(numerator / ((denominator_x * denominator_y) ** 0.5))

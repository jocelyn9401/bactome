32 def restriction_digest(seq, enzyme, max_band=23130, min_band=2000,
33  linear=False, p='yes'):
34  """
35  Performs restriction endonuclease digestion on a sequence and group the
36  resulting fragments into 3 groups so simulate different agarose gel
37  electrophoresis:
38  1. fragment length more than the maximum size
39  2. fragment length between the maximum and minimum size
40  3. fragment length less than the minimum size
41 
42  Parameters:
43  seq = DNA sequence for restriction endonuclease digestion
44  enzyme = Restriction endonuclease object from Bio.Restriction
45  package
46  max_band = size of maximum band in basepairs. Default = 23130
47  min_band = size of minimum band in basepairs. Default = 2000
48  linear = flag to define if DNA sequence is linear.
49  Default = False (DNA is circular)
50  p = flag to determine if the data is to be printed. Default = yes
51 
52  Result:
53  (Number of fragments after digestion,
54  List of fragments with molecular size above max_band,
55  List of fragments with molecular size between max_band and min_band,
56  List of fragments with molecular size below min_band)
57  """
58  digest = enzyme.search(seq, linear=linear)
59  digest.sort()
60  fragment = [digest[x+1] - digest[x]
61  for x in range(len(digest) - 1)]
62  fragment.sort()
63  ogel = [x for x in fragment if x > max_band]
64  gel = [x for x in fragment if x <= max_band and x >= min_band]
65  ugel = [x for x in fragment if x < min_band]
66  ogel.sort()
67  gel.sort()
68  ugel.sort()
69  if p == 'yes':
70  print('Enzyme: ' + str(enzyme))
71  print('Restriction site: ' + enzyme.site)
72  print('Number of fragments: ' + str(len(fragment)))
73  print('Number of fragments (x > ' + str(max_band) + '): ' + \
74  str(len(ogel)))
75  print('Number of fragments (' + str(max_band) + ' < x < ' + \
76  str(min_band) + '): ' + str(len(gel)))
77  print('Number of fragments (x < ' + str(min_band) + '): ' + \
78  str(len(ugel)))
79  print
80  return (len(fragment), ogel, gel, ugel)

142 def setCompare(original, test, absent):
143  """
144  Used for processing set-based (unordered or nominal) distance of
145  categorical data.
146 
147  Parameters:
148  original: list of original data
149  test: list of data to test against original
150  absent: indicator to define absent data
151  """
152  original_only = float(len([x for x in original
153  if x not in test]))
154  test_only = float(len([x for x in test if x not in original]))
155  both = float(len([x for x in original if x in test]))
156  return (original_only, test_only, both)

158 def listCompare(original, test, absent):
159  """
160  Used for processing list-based (ordered or ordinal) distance of
161  categorical data.
162 
163  Parameters:
164  original: list of original data
165  test: list of data to test against original
166  absent: indicator to define absent data
167  """
168  original = list(original)
169  test = list(test)
170  original_only = 0.0
171  test_only = 0.0
172  both = 0.0
173  for i in range(len(original)):
174  if original[i] == absent and test[i] == absent: pass
175  elif original[i] == test[i]: both = both + 1
176  elif original[i] != absent and test[i] == absent:
177  original_only = original_only + 1
178  elif original[i] == absent and test[i] != absent:
179  test_only = test_only + 1
180  else: pass
181  return (original_only, test_only, both)

183 def Nei_Li(original, test, absent=0, type='Set'):
184  """
185  Nei and Li Distance is distance measure for nominal or ordinal data.
186 
187  Given 2 lists (original and test), calculates the Nei and Li Distance
188  based on the formula,
189 
190  1 - [2 x (number of regions where both species are present)/
191  [(2 x (number of regions where both species are present)) +
192  (number of regions where only one species is present)]]
193 
194  @see: Nei M, Li WH (1979) Mathematical models for studying genetic
195  variation in terms of restriction endonucleases.
196  Proc Natl Acad Sci USA 76:5269-5273
197 
198  @param original: list of original data
199  @param test: list of data to test against original
200  @param absent: user-defined identifier for absent of region, default = 0
201  @param type: {Set | List}, define whether use Set comparison (unordered) or
202  list comparison (ordered), default = Set
203  """
204  if type == 'Set':
205  (original_only, test_only, both) = setCompare(original, test, absent)
206  else:
207  (original_only, test_only, both) = listCompare(original, test, absent)
208  return 1-((2*both)/((2*both)+original_only+test_only))

32 def compare(original, test, absent, type='Set'):
33  """
34  Used for processing list-based (positional) or set-based
35  (non-positional) distance of categorical data.
36 
37  @param original: list of original data
38  @param test: list of data to test against original
39  @param absent: indicator to define absent data
40  @param type: {List | Set}
41  """
42  if type == 'Set':
43  original_only = float(len([x for x in original if x not in test]))
44  test_only = float(len([x for x in test if x not in original]))
45  both = float(len([x for x in original if x in test]))
46  return (original_only, test_only, both, 0.0)
47  if type == 'List':
48  original, test = list(original), list(test)
49  original_only, test_only, both, none = 0.0, 0.0, 0.0, 0.0
50  for i in range(len(original)):
51  if original[i] == absent and test[i] == absent:
52  none = none + 1
53  elif original[i] == test[i]:
54  both = both + 1
55  elif original[i] <> absent and test[i] == absent:
56  original_only = original_only + 1
57  elif original[i] == absent and test[i] <> absent:
58  test_only = test_only + 1
59  else: pass
60  return (original_only, test_only, both, none)

605 def Tulloss(original, test, absent=0, type='Set'):
606  """
607  Tulloss coefficient for nominal or ordinal data.
608 
609  Coefficient:
610  M{sqrt(U * S * R)}
611  - M{U = log(1 + ((min(B, C) + A) / (max(B, C) + A))) / log2}
612  - M{S = 1 / sqrt(log(2 + (min(B, C) / (A + 1))) / log2)}
613  - M{R = log(1 + A / (A + B))log(1 + A / (A + C)) / log2log2}
614 
615  @param original: list of original data
616  @param test: list of data to test against original
617  @param absent: user-defined identifier for absent of region,
618  default = 0
619  @param type: {Set | List}, define whether use Set comparison
620  (non-positional) or list comparison (positional), default = Set
621 
622  @see: Ling, MHT. 2010. COPADS, I: Distances Measures between Two
623  Lists or Sets. The Python Papers Source Codes 2:2.
624  """
625  (original, test, both, none) = compare(original, test, absent, type)
626  U = (min(test, original) + both) / (max(test, original) + both)
627  U = math.log10(1 + U) / math.log10(2)
628  S = math.log10(2 + (min(test, original) / (both + 1)))
629  S = 1 / ((S / math.log10(2)) ** 0.5)
630  R = math.log10(1 + (both / (both + original)))
631  R = R * math.log10(1 + (both / (both + test)))
632  R = R / (math.log10(2) * math.log10(2))
633  return (U * S * R) ** 0.5

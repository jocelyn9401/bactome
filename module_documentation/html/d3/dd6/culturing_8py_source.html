27 def generation(A2, A1, min2, min1,
28  conv=50000000/0.3,
29  limit=0.301,
30  gradient=52137400,
31  intercept=118718650):
32  """
33  Calculates the number of generations and generation time using
34  turbidity measurement of cell culture (usually OD600 readings)
35  between 2 time points (min1 and min2).
36 
37  Parameters:
38  A2 = turbidity measurement at min2.
39  A1 = turbidity measurement at min1.
40  conv = conversion ratio between OD readings to cell count.
41  Default is 0.3 OD = 5e7 (50 million) cells, standard for
42  Escherichia coli.
43  limit = OD reading at which limit of proportionality of the
44  conversion ratio no longer holds, such as cell size decrease
45  or increase with higher OD readings.
46  Default is 0.301, for Escherichia coli.
47  gradient and intercept = the gradient and intercept for cell
48  number correction when OD is above limit of proportionality.
49  Correction equation:
50  cell count = gradient * ln(OD) + intercept
51  Default value of gradient is 52137400 and intercept is
52  118718650.
53 
54  Limit of proportionality, correction gradient and intercept
55  are reverse engineered from the graph of Sezonov, Guennadi,
56  Joseleau-Petit, Daniele and D'Ari, Richard. 2007. Escherichia
57  coli Physiology in Luria-Bertani Broth. Journal of Bacteriology
58  189(23):8746-8749
59 
60  Returns:
61  (number of generations between min1 and min2,
62  length of time (in minutes) for each generation (generation time),
63  number of cells per ml at min2,
64  number of cells per ml at min1)
65  """
66  if A1 < limit:
67  y = A1 * conv
68  else:
69  y = gradient * math.log(A1, math.e) + intercept
70  if A2 < limit:
71  x = A2 * conv
72  else:
73  x = gradient * math.log(A2, math.e) + intercept
74  gen = (math.log10(x) - math.log10(y))/math.log10(2)
75  time_diff = min2 - min1
76  return (gen, time_diff/gen, x, y)

78 def totalgen(ODs, p='yes', dilution=100,
79  conv=50000000/0.3, limit=0.301,
80  gradient=52137400, intercept=118718650):
81  """
82  Parameters:
83  ODs = list of OD readings at the start of next subculture.
84  p = flag to determine if the data is to be printed. Default = yes
85  dilution = dilution factor between each passage. Default = 100
86  conv = conversion ratio between OD readings to cell count.
87  Default is 0.3 OD = 5e7 (50 million) cells, standard for
88  Escherichia coli.
89  limit = OD reading at which limit of proportionality of the
90  conversion ratio no longer holds, such as cell size decrease
91  or increase with higher OD readings.
92  Default is 0.301, for Escherichia coli.
93  gradient and intercept = the gradient and intercept for cell
94  number correction when OD is above limit of proportionality.
95  Correction equation:
96  cell count = gradient * ln(OD) + intercept
97  Default value of gradient is 52137400 and intercept is
98  118718650.
99 
100  Limit of proportionality, correction gradient and intercept
101  are reverse engineered from the graph of Sezonov, Guennadi,
102  Joseleau-Petit, Daniele and D'Ari, Richard. 2007. Escherichia
103  coli Physiology in Luria-Bertani Broth. Journal of Bacteriology
104  189(23):8746-8749
105  """
106  ODs = [float(x) for x in ODs if x != '']
107  gen_list = [0] * (len(ODs) - 1)
108  for passage in range(len(ODs)-1):
109  if ODs[passage] < limit:
110  initial_count = ODs[passage] * conv
111  else:
112  initial_count = gradient * math.log(ODs[passage], math.e) + \
113  intercept
114  if ODs[passage+1] < limit:
115  final_count = ODs[passage+1] * limit
116  else:
117  final_count = gradient * math.log(ODs[passage+1], math.e) + \
118  intercept
119  initial_count = initial_count / float(dilution)
120  gen = (math.log10(final_count) - \
121  math.log10(initial_count)) / math.log10(2)
122  if gen < 0: gen = 0.0
123  gen_list[passage] = gen
124  if p == 'yes':
125  print 'Number of cells inoculate from Passage ' + \
126  str(passage) + ' = ' +str(initial_count)
127  print 'Number of cells at subculture from Passage ' + \
128  str(passage+1) + ' = ' +str(final_count)
129  print 'Between Passage ' + str(passage) + ' and Passage ' + \
130  str(passage+1) + ', Number of generations = ' + \
131  str(gen_list[passage])
132  print 'Total number of generations since Passage 0 = ' + \
133  str(sum(gen_list))
134  print
135  return gen_list

31 def parse_GAF(gaf_file):
32  """
33  Reader for GO Annotation File - http://www.geneontology.org/
34  GO.format.gaf-2_0.shtml
35 
36  @param gaf_file: file path for GO annotation file
37  @returns: a tuple - (gaf, columns, gaf_ver, header) -
38  "gaf" is the annotation,
39  "columns" is the column header in the annotation,
40  "gaf_ver" is the GAF version (either 1.0 or 2.0),
41  "header" contains all the comments in the annotation file.
42 
43  @raise AttributeError: when there is no GAF version or if GAF
44  version is not 1.0 or 2.0
45  """
46  gaf = open(gaf_file, 'r').readlines()
47  gaf = [x[:-1].strip() for x in gaf]
48  header = [x for x in gaf if x.startswith('!')]
49  gaf = [x for x in gaf if not x.startswith('!')]
50  if '!gaf-version: 2.0' in header:
51  gaf_ver = 2.0
52  elif '!gaf-version: 1.0' in header:
53  gaf_ver = 1.0
54  else:
55  raise AttributeError('File does not have GAF version')
56  gaf = [x.split('\t') for x in gaf]
57  gaf_16 = [x for x in gaf if len(x) == 16]
58  gaf_17 = [x for x in gaf if len(x) == 17]
59  gaf = [x for x in gaf if len(x) == 15]
60  gaf = [(x[0].strip(), x[1].strip(), x[2].strip(),
61  [y.strip() for y in x[3].split('|')], x[4].strip(),
62  [y.strip() for y in x[5].split('|')], x[6].strip(),
63  [y.strip() for y in x[7].split('|')], x[8].strip(),
64  x[9].strip(), [y.strip() for y in x[10].split('|')],
65  x[11].strip(), [y.strip() for y in x[12].split('|')],
66  x[13].strip(), x[14].strip())
67  for x in gaf]
68  gaf_16 = [(x[0].strip(), x[1].strip(), x[2].strip(),
69  [y.strip() for y in x[3].split('|')], x[4].strip(),
70  [y.strip() for y in x[5].split('|')], x[6].strip(),
71  [y.strip() for y in x[7].split('|')], x[8].strip(),
72  x[9].strip(), [y.strip() for y in x[10].split('|')],
73  x[11].strip(), [y.strip() for y in x[12].split('|')],
74  x[13].strip(), x[14].strip(),
75  [y.strip() for y in x[15].split('|')])
76  for x in gaf_16]
77  gaf_17 = [(x[0].strip(), x[1].strip(), x[2].strip(),
78  [y.strip() for y in x[3].split('|')], x[4].strip(),
79  [y.strip() for y in x[5].split('|')], x[6].strip(),
80  [y.strip() for y in x[7].split('|')], x[8].strip(),
81  x[9].strip(), [y.strip() for y in x[10].split('|')],
82  x[11].strip(), [y.strip() for y in x[12].split('|')],
83  x[13].strip(), x[14].strip(),
84  [y.strip() for y in x[15].split('|')],
85  [y.strip() for y in x[16].split('|')])
86  for x in gaf_17]
87  gaf = gaf + gaf_16 + gaf_17
88  if gaf_ver == 2.0:
89  columns = ['DB', 'DB Object ID', 'DB Object Symbol',
90  'Qualifier', 'GO ID', 'DB:Reference', 'Evidence Code',
91  'With (or) From', 'Aspect', 'DB Object Name',
92  'DB Object Synonym', 'DB Object Type', 'Taxon',
93  'Date', 'Assigned By', 'Annotation Extension',
94  'Gene Product Form ID']
95  if gaf_ver == 1.0:
96  columns = ['DB', 'DB Object ID', 'DB Object Symbol',
97  'Qualifier', 'GO ID', 'DB:Reference', 'Evidence Code',
98  'With (or) From', 'Aspect', 'DB Object Name',
99  'DB Object Synonym', 'DB Object Type', 'Taxon',
100  'Date', 'Assigned By']
101  return (gaf, columns, gaf_ver, header)

103 def process_association_file(assoc_file):
104  """
105  Extracts information from GO annotation file for over-representation
106  analysis
107 
108  @param assoc_file: File path for GO annotation file
109 
110  @return: A tuple - (genome, gene_set, length of gene_set,
111  ontology_set, length of ontology_set) -
112  "genome" is a list of (<gene symbol>, <GO ID>) representing the
113  annotated genome
114  "gene_set" is the set of genes in the genome
115  "length of gene_set" is the number of genes in the genome
116  "ontology_set" is the set of GO IDs used in the annotation
117  "length of ontology_set" is the number of GO IDs used in the
118  annotation
119  """
120  genome = parse_GAF(assoc_file)[0]
121  genome = [(x[2].lower(), x[4].lower())
122  for x in genome
123  if len(x) > 5]
124  gene_set = set([x[0] for x in genome])
125  ontology_set = set([x[1] for x in genome])
126  return (genome, gene_set, len(gene_set), ontology_set,
127  len(ontology_set))

129 def process_sample_file(samplefile, genome):
130  """
131  Processes the gene list file, the set of genes to be analyzed for
132  GO over-representation, and generate an annotated sample genome
133  (to represent the observed data in Chi Square test)
134 
135  @param samplefile: File path for the gene list for over-representation
136  analysis - one gene per line
137  @param genome: Genome to base expectation on. The genome is a list
138  of (<gene symbol>, <GO ID>)
139  """
140  sample_list = open(samplefile, 'r').readlines()
141  sample_list = [x[:-1].lower() for x in sample_list]
142  sample_genome = [x for x in genome
143  if x[0].lower() in sample_list]
144  gene_set = set([x[0] for x in sample_genome])
145  sample_ontology = set([x[1] for x in sample_genome])
146  return(sample_list, sample_genome, gene_set, len(gene_set),
147  sample_ontology)

173 def calculate_expectation(ontology, genome, genome_count):
174  """
175  Calculated the expectation (number of expected in Chi Square test)
176  for each ontology ID and the given genome
177 
178  @param ontology: Gene ontology ID to calculate expectation
179  @param genome: Genome to base expectation on. The genome is a list
180  of (<gene symbol>, <GO ID>)
181  @param genome_count: Number of genes in the genome
182 
183  @return: Expectation as a float number.
184  """
185  gene_w_ontology = len(find_gene_by_ontology(ontology, genome))
186  return float(gene_w_ontology) / float(genome_count)

236 def main(assoc_file, samplefile, outputfile, display=True):
237  """
238  Runner for the GO over-representation analysis
239 
240  @param assoc_file: File path for GO annotation file
241  @param samplefile: File path for the gene list for over-representation
242  analysis - one gene per line
243  @param outputfile: File path to write the analysis output
244  @param display: Flag to display the results on console (default = True)
245  """
246  (genome, gene_set, genome_count, ontology_set, ontology_count) = \
247  process_association_file(assoc_file)
248  (sample_list, sample_genome, gene_set, sample_count, sample_ontology) = \
249  process_sample_file(samplefile, genome)
250  output_f = open(outputfile, 'w')
251  genes_not_mapped = ', '.join([x for x in sample_list
252  if x not in gene_set])
253  genes_mapped = ', '.join([x for x in sample_list
254  if x in gene_set])
255  _write_headers(output_f, assoc_file, samplefile, outputfile,
256  genome_count, ontology_count, len(sample_list),
257  len(gene_set), genes_not_mapped, genes_mapped,
258  len(sample_ontology))
259  output_f.write(','.join(['Term', 'Expected Count', '(Not) Expected',
260  'Actual Count', '(Not) Actual', '\n']))
261  if display:
262  print('\t'.join(['Term', 'Expected Count', '(Not) Expected',
263  'Actual Count', '(Not) Actual']))
264  for ontology in sample_ontology:
265  expectation = calculate_expectation(ontology, genome, genome_count)
266  expected_count = expectation * sample_count
267  expected_uncount = sample_count - expected_count
268  actual_count = len(find_gene_by_ontology(ontology, sample_genome))
269  actual_uncount = sample_count - actual_count
270  output_f.write(','.join([ontology,
271  str(expected_count),
272  str(expected_uncount),
273  str(actual_count),
274  str(actual_uncount), '\n']))
275  if display:
276  print('\t'.join([ontology,
277  str(expected_count),
278  str(expected_uncount),
279  str(actual_count),
280  str(actual_uncount)]))
281  output_f.close()

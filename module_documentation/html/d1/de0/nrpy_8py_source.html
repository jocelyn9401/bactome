378 def betacf(a, b, x):
379  """
380  Continued fraction for incomplete beta function.
381  Adapted from salstat_stats.py of SalStat (www.sf.net/projects/salstat)
382  @see: NRP 6.3
383  @see: Ling, MHT. 2009. Compendium of Distributions, I: Beta, Binomial, Chi-
384  Square, F, Gamma, Geometric, Poisson, Student's t, and Uniform. The Python
385  Papers Source Codes 1:4
386  """
387  iter_max = 200
388  eps = 3.0e-7
389 
390  bm = az = am = 1.0
391  qab = a + b
392  qap = a + 1.0
393  qam = a - 1.0
394  bz = 1.0 - qab * x / qap
395  for i in range(iter_max + 1):
396  em = float(i + 1)
397  tem = em + em
398  d = em * (b - em) * x / ((qam + tem) * (a + tem))
399  ap = az + d * am
400  bp = bz + d * bm
401  d = -(a + em) * (qab + em) * x / ((qap + tem) * (a + tem))
402  app = ap + d * az
403  bpp = bp + d * bz
404  aold = az
405  am = ap / bpp
406  bm = bp / bpp
407  az = app / bpp
408  bz = 1.0
409  if (abs(az - aold) < (eps * abs(az))):
410  return az

412 def betai(a, b, x):
413  """
414  Incomplete beta function
415 
416  I-sub-x(a,b) = 1/B(a,b)*(Integral(0,x) of t^(a-1)(1-t)^(b-1) dt)
417 
418  where a,b>0 and B(a,b) = G(a)*G(b)/(G(a+b)) where G(a) is the gamma
419  function of a.
420 
421  Adapted from salstat_stats.py of SalStat (www.sf.net/projects/salstat)
422  Depend: betacf, gammln
423  @see: NRP 6.3
424 
425  @see: Ling, MHT. 2009. Compendium of Distributions, I: Beta, Binomial, Chi-
426  Square, F, Gamma, Geometric, Poisson, Student's t, and Uniform. The Python
427  Papers Source Codes 1:4
428  """
429  if (x < 0.0 or x > 1.0):
430  raise ValueError('Bad value for x: %s' % x)
431  if (x == 0.0 or x == 1.0):
432  bt = 0.0
433  else:
434  bt = math.exp(gammln(a+b) - gammln(a) - gammln(b) + a *
435  math.log(x) + b * math.log(1.0-x))
436  if (x < (a + 1.0) / (a + b + 2.0)):
437  return bt * betacf(a, b, x) / float(a)
438  else:
439  return 1.0 - bt * betacf(b, a, 1.0 - x) / float(b)

627  b1 = 1.0

631  an = n

637  b1 = x * b0 + anf * b1

12 class World(object):
13  '''
14  Representation of a 3-dimensional ecological world.
15 
16  The ecosystem is made up of ecological cells. Each ecological cell is
17  modelled as a dictionary of
18  - local_input: A list containing processed input, representing the partial
19  local ecological condition, to be used as input to the organisms in the
20  current ecological cell. This is updated by World.update_local function.
21  - local_output: A list containing processed output, representing the partial
22  local ecological condition. This is updated by World.update_local
23  function.
24  - temporary_input: A list acting as temporary holding for input after being
25  fed to the organisms in the current ecological cell, which is to be used
26  to update local_input and local_output lists by World.update_local and
27  World.update_ecology functions.
28  - temporary_output: A list acting as temporary holding for output from the
29  organisms in the current ecological cell, which is to be used to update
30  local_input and local_output lists by World.update_local and
31  World.update_ecology functions.
32  - organisms: The number of organisms in the current ecological cell which
33  is updated by World.organism_movement and World.organism_location
34  functions.
35 
36  @see: Ling, MHT. 2012. An Artificial Life Simulation Library Based on
37  Genetic Algorithm, 3-Character Genetic Code and Biological Hierarchy. The
38  Python Papers 7: 5.
39  '''

42  def __init__(self, world_x, world_y, world_z):
43  '''
44  Setting up the world and ecosystem
45 
46  @param world_x: number of ecological cells on the x-axis
47  @type world_x: integer
48  @param world_y: number of ecological cells on the y-axis
49  @type world_y: integer
50  @param world_z: number of ecological cells on the z-axis
51  @type world_z: integer
52  '''
53  eco_cell = {'local_input': [], 'local_output': [],
54  'temporary_input': [], 'temporary_output': [],
55  'organisms': 0}
56  self.world_x = int(world_x)
57  self.world_y = int(world_y)
58  self.world_z = int(world_z)
59  for x in range(self.world_x):
60  eco_x = {}
61  for y in range(self.world_y):
62  eco_y = {}
63  for z in range(self.world_z): eco_y[z] = copy.deepcopy(eco_cell)
64  eco_x[y] = copy.deepcopy(eco_y)
65  self.ecosystem[x] = copy.deepcopy(eco_x)

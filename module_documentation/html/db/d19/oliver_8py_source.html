43 def datafile(filename):
44  '''
45  Reads input data file (containing probe and expression data) into a
46  dictionary.
47 
48  Input data file is a comma-delimited file in the format of
49  ProbeName1,Sample1Value,Sample2value,Sample3Value, ...
50  ProbeName2,Sample1Value,Sample2value,Sample3Value, ...
51  ProbeName3,Sample1Value,Sample2value,Sample3Value, ...
52 
53  @param filename: input data file name
54  @return: dictionary where key is ProbeName and value is a list of
55  SampleValues
56  '''
57  print 'Reading data from', filename
58  t = time.time()
59  f = open(filename, 'r').readlines()
60  if len(f) == 1:
61  f = f[0].split('\r')
62  f = [x[:-1] for x in f]
63  f = [x.split(',') for x in f]
64  data = {}
65  for x in f:
66  try:
67  x = [y for y in x if y != '']
68  if len(x) > 2:
69  name = str(x[0])
70  values = [float(y) for y in x[1:]]
71  data[name] = values
72  except: 'Error formatting:', x
73  print ' Time taken:', time.time()-t, 'seconds'
74  return data

259 def p_geomean_expratio_cv(data, options, methods, results):
260  '''
261  Processor for reference gene identification method: geometric mean
262  of exponent of ratio correlation (e^ratio) and coefficient of variation
263 
264  @param data: dictionary of input data from datafile function
265  @param options: options for current method (please see user manual)
266  @param methods: list of methods used (current method will be appended
267  for final result file)
268  @param results: results dictionary in the format of {<ProbeName>:
269  {<method>: <results from method>}}
270  @return: tuple of (methods, results)
271  '''
272  print 'Calculating geometric mean of ratio correlation and CV ...'
273  (methods, results) = p_selfed_ratio_correlation(data, options,
274  methods, results)
275  (methods, results) = p_cv(data, options, methods, results)
276  methods.append('geomean_expratio_cv')
277  for genename in results.keys():
278  eratio = math.e ** results[genename]['ratio_corr']
279  cv = results[genename]['cv']
280  geomean = math.sqrt(eratio * cv)
281  if results.has_key(genename):
282  results[genename]['geomean_expratio_cv'] = geomean
283  else:
284  results[genename] = {'geomean_expratio_cv': geomean}

287 def p_avgexpratio_avgcv(data, options, methods, results):
288  '''
289  Processor for reference gene identification method:
290  (e^ratio_correlation / average of e^ratio_correlation) +
291  (cv / average of cv)
292 
293  @param data: dictionary of input data from datafile function
294  @param options: options for current method (please see user manual)
295  @param methods: list of methods used (current method will be appended
296  for final result file)
297  @param results: results dictionary in the format of {<ProbeName>:
298  {<method>: <results from method>}}
299  @return: tuple of (methods, results)
300  '''
301  print 'Calculating sum of averaged ratio correlation and averaged CV ...'
302  (methods, results) = p_selfed_ratio_correlation(data, options,
303  methods, results)
304  (methods, results) = p_cv(data, options, methods, results)
305  methods.append('avgexpratio_avgcv')
306  avgexpratio = [math.e ** results[genename]['ratio_corr']
307  for genename in results.keys()]
308  avgexpratio = sum(avgexpratio) / float(len(avgexpratio))
309  avgcv = [results[genename]['cv']
310  for genename in results.keys()]
311  avgcv = sum(avgcv) / float(len(avgcv))
312  for genename in results.keys():
313  eratio = math.e ** results[genename]['ratio_corr']
314  cv = results[genename]['cv']
315  r = (float(eratio) / avgexpratio) + (float(cv) / avgcv)
316  if results.has_key(genename):
317  results[genename]['avgexpratio_avgcv'] = r
318  else:
319  results[genename] = {'avgexpratio_avgcv': r}
320  return (methods, results)

322 def processor(data, options):
323  '''
324  Main processor loop to execute each required reference gene
325  identification method processors
326 
327  @param data: dictionary of input data from datafile function
328  @param options: options for current method (please see user manual)
329  @return: tuple of (methods, results) where methods is list of methods
330  used, and results is a dictionary in the format of {<ProbeName>:
331  {<method>: <results from method>}}
332  '''
333  results = {}
334  methods = []
335  if options.has_key('all_methods') or options.has_key('gradient'):
336  # Method: gradient
337  (methods, results) = p_gradient(data, options, methods, results)
338  if options.has_key('all_methods') or options.has_key('cv'):
339  # Method: CV
340  (methods, results) = p_cv(data, options, methods, results)
341  if options.has_key('all_methods') or options.has_key('regression'):
342  # Method: regression ratios (R^2/slope)
343  (methods, results) = p_regression_ratio(data, options,
344  methods, results)
345  if options.has_key('all_methods') or options.has_key('avgstd'):
346  # Method: average x stdev
347  (methods, results) = p_average_stdev(data, options,
348  methods, results)
349  if options.has_key('all_methods') or options.has_key('pcorr'):
350  # Method: selfed product correlation
351  (methods, results) = p_selfed_product_correlation(data, options,
352  methods, results)
353  if options.has_key('all_methods') or options.has_key('rcorr'):
354  # Method: selfed ratio correlation
355  (methods, results) = p_selfed_ratio_correlation(data, options,
356  methods, results)
357  if options.has_key('all_methods') or options.has_key('scorr'):
358  # Method: selfed correlation
359  (methods, results) = p_selfed_correlation(data, options,
360  methods, results)
361  # Method: geomean(e^ratio, CV)
362  (methods, results) = p_geomean_expratio_cv(data, options,
363  methods, results)
364  # Method: (e^ratio/avg(e^ratio)) + (cv/avg(cv))
365  (methods, results) = p_avgexpratio_avgcv(data, options,
366  methods, results)
367  return (methods, results)

148  def Dijkstra(self, start, end=None):
149  """
150  Find shortest paths from the start vertex to all vertices nearer than
151  or equal to the end.
152 
153  Dijkstra's algorithm is only guaranteed to work correctly when all
154  edge lengths are positive. This code does not verify this property
155  for all edges (only the edges seen before the end vertex is reached),
156  but will correctly compute shortest paths even for some graphs with
157  negative edges, and will raise an exception if it discovers that a
158  negative edge has caused it to make a mistake.
159  """
160  D = {} # dictionary of final distances
161  P = {} # dictionary of predecessors
162  Q = PriorityDictionary() # est.dist. of non-final vert.
163  Q[start] = 0
164  for v in Q:
165  D[v] = Q[v]
166  if v == end: break
167  try: self.graph[v]
168  except KeyError: continue
169  for w in self.graph[v]:
170  if self.graph[v][w] > 0:
171  vwLength = D[v] + self.graph[v][w]
172  if w in D:
173  if vwLength < D[w]:
174  raise ValueError("Dijkstra: found better path to \
175  already-final vertex")
176  elif w not in Q or vwLength < Q[w]:
177  Q[w] = vwLength
178  P[w] = v
179  return (D, P)

182  """
183  Find a single shortest path from the given start vertex

185  in order along the shortest path.
186

188  @param end: vertex of ending point
189  """

192  while 1:
193  Path.append(end)

195  end = P[end]

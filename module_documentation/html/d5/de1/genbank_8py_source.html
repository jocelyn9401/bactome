27 class GenBankFile(object):
28  '''!
29  Class to read a Genbank file, parse it using BioPython library,
30  into a Python dictionary. If the Genbank file consists of more
31  than one Genbank records, the respective Genbank record name will
32  be used as key in the dictionary.
33 
34  A Genbank record consists of the following keys (as described in
35  https://www.ncbi.nlm.nih.gov/Sitemap/samplerecord.html):
36  - 'name' is the Genbank record name (usually accession number)
37  - 'description'
38  - 'accessions'
39  - 'comment'
40  - 'data_file_division'
41  - 'date'
42  - 'features' is a list of feature dictionary
43  - 'keywords'
44  - 'organism'
45  - 'sequence'
46  - 'sequence_version'
47  - 'source'
48  - 'taxonomy'
49  - 'topology'
50 
51  where feature dictionary consists of the following keys:
52  - 'id'
53  - 'start'
54  - 'end'
55  - 'type'
56  - 'strand'
57  '''

65  def _read_gb_features(self, gb_record):
66  '''!
67  Private method - used to read the features of a Genbank
68  record.
69  '''
70  features = [0] * len(gb_record.features)
71  for i in range(len(gb_record.features)):
72  f = gb_record.features[i]
73  record = {'type': f.type,
74  'start': f.location.start.position,
75  'end': f.location.end.position}
76  try: record['id'] = f.id
77  except: pass
78  try: record['strand'] = f.strand
79  except: pass
80  for key in f.qualifiers.keys():
81  record[key] = f.qualifiers[key]
82  features[i] = record
83  return features

85  def _read_gb_annotations(self, record, gb_record):
86  '''!
87  Private method - used to read the annotations of a Genbank
88  record.
89  '''
90  annotations = gb_record.annotations
91  try:
92  record['accessions'] = annotations['accessions']
93  except KeyError: pass
94  try:
95  comment = annotations['comment'].split('\n')
96  record['comment'] = '|'.join([x.strip() for x in comment])
97  except KeyError: pass
98  try:
99  record['data_file_division'] = annotations['data_file_division']
100  except KeyError: pass
101  try:
102  record['date'] = annotations['date']
103  except KeyError: pass
104  try:
105  record['keywords'] = annotations['keywords']
106  except KeyError: pass
107  try:
108  record['organism'] = annotations['organism']
109  except KeyError: pass
110  try:
111  record['sequence_version'] = annotations['sequence_version']
112  except KeyError: pass
113  try:
114  record['source'] = annotations['source']
115  except KeyError: pass
116  try:
117  record['taxonomy'] = annotations['taxonomy']
118  except KeyError: pass
119  try:
120  record['topology'] = annotations['topology']
121  except KeyError: pass
122  return record

124  def readGB(self, filepath):
125  '''!
126  Method to read a GenBank file and parse it into a dictionary.
127 
128  @param filepath string: file path of the Genbank file
129  '''
130  self.filepath = filepath
131  gb = open(self.filepath, 'r')
132  self.records = {}
133  for gb_record in SeqIO.parse(gb, 'genbank'):
134  record = {'name': gb_record.name,
135  'description': gb_record.description}
136  record['features'] = self._read_gb_features(gb_record)
137  record['sequence'] = str(gb_record.seq)
138  record = self._read_gb_annotations(record, gb_record)
139  self.records[gb_record.name] = record

299 def recordSelector(gbfile, RecordIndex=0, RecordID=None,
300  returntype='sequence'):
301  '''!
302  Function to select a Genbank record in a multi-record Genbank
303  file.
304 
305  @param gbfile string: file path of the Genbank file
306  @param RecordIndex Integer: Relative Genbank record number in
307  the given Genbank file where 0 is the first record and 1 is
308  the secord record, and so on
309  @param RecordID String: Unique name (identifying IDs) for the
310  specific Genbank record
311  @param returntype String: Defines the data to return; currently,
312  only 'sequence' and 'features' are acceptable
313  '''
314  gb = GenBankFile()
315  gb.readGB(gbfile)
316  IDs = list(gb.getIDs())
317  if RecordIndex == 0 or RecordID == None:
318  ID = IDs[0]
319  elif RecordIndex != 0 and RecordID == None:
320  ID = int(RecordIndex)
321  elif RecordID != None:
322  ID = RecordIndex
323  if returntype == 'sequence':
324  return gb.getSequence(ID)
325  elif returntype == 'features':
326  return gb.getFeatures(ID)

360 def displayFeatureTypes(gbfile='DSM6083.gb'):
361  '''!
362  Function to read a Genbank file and prints out the number of each
363  feature for each Genbank record in the file.
364 
365  Usage:
366 
367  python genbank.py displayfeaturetypes --gbfile=<Genbank
368  file path>
369 
370  The format of the output is
371 
372  <feature type> : <count>
373 
374  where count is the number of features for each feature type.
375 
376  @param gbfile string: file path of the Genbank file
377  '''
378  gb = GenBankFile()
379  gb.readGB(gbfile)
380  print('Number of Records: ' + str(len(gb.getIDs())))
381  print()
382  for ID in list(gb.getIDs()):
383  features = gb.getFeatures(ID)
384  count = {}
385  for f in features:
386  if f['type'] not in count:
387  count[f['type']] = 1
388  else:
389  count[f['type']] = count[f['type']] + 1
390  print('Name: ' + gb.getName(ID))
391  for k in count:
392  print('%s : %i' % (k, count[k]))
393  print()

395 def featureMap(feature='CDS', RecordIndex=0, RecordID=None,
396  gbfile='DSM6083.gb', expand=False, resolution=1):
397  '''!
398  Function to print out the base position(s) or base range for
399  a specific feature type in a specific Genbank record within
400  the given Genbank file.
401 
402  Usage 1: Prints the range of bases for a required feature
403 
404  python genbank.py featuring --feature=<feature type>
405  --RecordIndex=<Record number> --gbfile=<Genbank file path>
406 
407  Usage 2: Prints out individual base position for a required
408  feature
409 
410  python genbank.py featuring --feature=<feature type>
411  --RecordIndex=<record number> --gbfile=<Genbank file path>
412  --expand=False --resolution=<resolution>
413 
414  where
415  - record number is the relative Genbank record number in
416  the given Genbank file where 0 is the first record and
417  1 is the secord record, and so on
418  - resolution is to control the resolution of printouts.
419  For example, resolution of 1 will print out every base
420  position for a given feature type; which will generate
421  a long printout. Increasing the resolution will reduce
422  the volume of printout but may not accurately pin-point
423  the end base position of the feature
424 
425  Usage 1 will generate the following output:
426 
427  <feature type> : <starting base position> : <ending
428  base position>
429 
430  Usage 2 will generate the following output:
431 
432  <feature type> : <base position>
433 
434  @param feature String: Type of feature
435  @param RecordIndex Integer: Relative Genbank record number in
436  the given Genbank file where 0 is the first record and 1 is
437  the secord record, and so on
438  @param RecordID String: Unique name (identifying IDs) for the
439  specific Genbank record
440  @param gbfile string: file path of the Genbank file
441  @param expand Boolean: Flag to define whether to expand from
442  start/end base positions to indivisual base positions
443  @param resolution Integer: Control the resolution of printouts.
444  For example, resolution of 1 will print out every base
445  position for a given feature type; which will generate a long
446  printout. Increasing the resolution will reduce the volume of
447  printout but may not accurately pin-point the end base position
448  of the feature
449  '''
450  features = recordSelector(gbfile, RecordIndex, RecordID,
451  'features')
452  required_feature = str(feature)
453  expand = str(expand).lower()
454  for f in features:
455  if f['type'] == required_feature:
456  if expand == 'false':
457  print('%s : %s : %s' % \
458  (feature, str(f['start']), str(f['end'])))
459  if expand == 'true':
460  resolution = int(resolution)
461  for base in range(f['start'], f['end']+1,
462  resolution):
463  print('%s : %s' % (feature, base))

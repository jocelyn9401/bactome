178  def __init__(self, location, scale, p, q):
179  """
180  Constructor method. The parameters are used to construct the
181  probability distribution.
182 
183  @param location:
184  @param scale: upper bound
185  @param p: shape parameter. Although no upper bound but seldom exceed 10.
186  @param q: shape parameter. Although no upper bound but seldom exceed 10.
187  """
188  self.location = float(location)
189  self.scale = float(scale)
190  self.p = float(p)
191  self.q = float(q)
192 
193  def CDF(self, x):
194  """
195  Cummulative Distribution Function, which gives the cummulative
196  probability (area under the probability curve) from -infinity or 0 to
197  a give x-value on the x-axis where y-axis is the probability.
198  """
199  return nrpy.betai(self.p, self.q, (x - self.location)/
200  (self.scale - self.location))
201 
202  def PDF(self, x):
203  """
204  Partial Distribution Function, which gives the probability
205  for particular value of x, or the area under probability
206  distribution from x-h to x+h for continuous distribution.
207  """
208  n = (self.scale - self.location) ** (self.p + self.q - 1)
209  n = nrpy.gammln(self.p) * nrpy.gammln(self.q) * n
210  n = nrpy.gammln(self.p + self.q) / n
211  p = (x - self.location) ** (self.p - 1)
212  q = (self.scale - x) ** (self.q - 1)
213  return n * p * q
214 
215  def inverseCDF(self, probability, start=0.0, step=0.01):
216  """
217  It does the reverse of CDF() method, it takes a probability
218  value and returns the corresponding value on the x-axis.
219  """
220  cprob = self.CDF(start)
221  if probability < cprob:
222  return (start, cprob)
223  while probability > cprob:
224  start = start + step
225  cprob = self.CDF(start)
226  return (start, cprob)
227 
228  def mean(self):
229  """Gives the arithmetic mean of the sample."""
230  n = (self.location * self.q) + (self.scale * self.p)
231  return n / (self.p + self.q)
232 
233  def mode(self):
234  """Gives the mode of the sample."""
235  n = (self.location * (self.q - 1)) + (self.scale * \
236  (self.p - 1))
237  return n / (self.p + self.q - 2)
238 
239  def kurtosis(self):
240  """Gives the kurtosis of the sample."""
241  n = (self.p ** 2) * (self.q + 2) + \
242  (2 * (self.q ** 2)) + \
243  ((self.p * self.q) * (self.q - 2))
244  n = n * (self.p + self.q + 1)
245  d = self.p * self.q * (self.p + self.q + 2) * \
246  (self.p + self.q + 3)
247  return 3 * ((n / d) - 1)
248 
249  def skew(self):
250  """Gives the skew of the sample."""
251  d = (self.p + self.q) ** 3
252  d = d * (self.p + self.q + 1) * (self.p + self.q + 2)
253  e = ((self.p + self.q) ** 2) * (self.p + self.q + 1)
254  e = (self.p * self.q) / e
255  e = e ** 1.5
256  return ((2 * self.p * self.q) * (self.q - self.q)) / (d * e)
257 
258  def variance(self):
259  """Gives the variance of the sample."""
260  n = self.p * self.q * ((self.scale - self.location) ** 2)
261  d = (self.p + self.q + 1) * ((self.p + self.q) ** 2)
262  return n / d
263 
264  def moment(self, r):
265  """Gives the r-th moment of the sample."""
266  return nrpy.beta(self.p + r,
267  self.q)/nrpy.beta(self.p, self.q)
268 
269  def random(self):
270  """Gives a random number based on the distribution."""
271  return random.betavariate(self.p, self.q)

668  probability (area under the probability curve) from -infinity or 0 to

670  """
671  return 1.0 - 0.5 * nrpy.erfcc(x/SQRT2)
672  def PDF(self, x):
673  """
674  Calculates the density (probability) at x by the formula
675  f(x) = 1/(sqrt(2 pi) sigma) e^-((x^2/(2 sigma^2))
676  where mu is the mean of the distribution and sigma the standard
677  deviation.
678

680  """
681  return (1/(math.sqrt(PI2) * self.stdev)) * \
682  math.exp(-(x ** 2/(2 * self.stdev**2)))
683  def inverseCDF(self, probability, start = -10.0,
684  end = 10.0, error = 10e-8):
685  """
686  It does the reverse of CDF() method, it takes a probability value and
687  returns the corresponding value on the x-axis, together with the
688  cumulative probability.
689 
690  @param probability: probability under the curve from -infinity
691  @param start: lower boundary of calculation (default = -10)
692  @param end: upper boundary of calculation (default = 10)
693  @param error: error between the given and calculated probabilities
694  (default = 10e-8)
695  @return: Returns a tuple (start, cprob) where 'start' is the standard
696  deviation for the area under the curve from -infinity to the given
697  'probability' (+/- step). 'cprob' is the calculated area under the
698  curve from -infinity to the returned 'start'.
699  """
